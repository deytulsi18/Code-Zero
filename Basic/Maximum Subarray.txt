class Solution {
    public int maxSubArray(int[] nums) {
        int maxSum = nums[0];
        int currSum = nums[0];

        for (int i = 1; i < nums.length; i++) {
            // currSum = currSum < 0 ? 0 : currSum;
            // currSum += nums[i];
            currSum = Math.max(currSum + nums[i], nums[i]);
            maxSum = maxSum < currSum ? currSum : maxSum;
        }

        return maxSum;
    }
}
