class Solution {
    public int removeElement(int[] nums, int val) {
        int pos_zero = 0;
        int pos_curr = 0;
        while (pos_curr < nums.length){
            if (nums[pos_curr] == val) {
                pos_curr++;
            } else {
                nums[pos_zero] = nums[pos_curr];
                pos_curr++;
                pos_zero++;
            }
        }

        return pos_zero;
    }
}
